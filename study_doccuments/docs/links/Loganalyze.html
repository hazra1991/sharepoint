Quick extra example: Weighted scoring rule engine sketch

python
class WeightedPromptSelector:
    def __init__(self):
        self.rules = []  # each rule: (weight, fn)

    def add_rule(self, weight, rule_fn):
        self.rules.append((weight, rule_fn))

    def select(self, prompts: list[Prompt], current_log: str) -> Prompt:
        scores = {p: 0 for p in prompts}
        for weight, rule in self.rules:
            results = rule(prompts, current_log)  # returns dict {prompt:score_delta}
            for p, score_delta in results.items():
                scores[p] += weight * score_delta
        return max(scores, key=scores.get)
